List

按照算法入门page50的代码复现了链表建立、插入操作。

#include <stdio.h>
#include <bits/stdc++.h>
using namespace std;
struct node
{
    int data;
    struct node* next;
};
int main()
{
    int a, num;
    struct node *head, *p, *q, *t;
    head = NULL;
    cin >> num; // original number of sequence 
    for(int i = 0; i < num; ++i)
    {
        p = (struct node*)malloc(sizeof(struct node)); // apply for a place of memory for node
        cin >> a;
        p -> data = a;
        p -> next = NULL;
        if(head == NULL) head = p; // head points at p
        else q->next = p; // make q points at p

        q = p; // q => p 
    }

    cin >> a; // the one to be inserted
    t = head;
    // p = head; // reserve the former one pointer
    while(t!= NULL)
    {
        if(t->next->data > a)  // find the one to be
        {
            p = (struct node*)malloc(sizeof(struct node));
            p->data = a;
            p->next = t->next;
            t->next = p;
           // free(p);  No need to free, We still need to use it.
            break;
        }
        else
            t = t->next;
    }

   // cout << head->data << endl;
    t = head;
    while(t != NULL)
    {
        cout << t->data << ' ';
        t = t->next;
    }

    return 0;
}

可惜这里代码是存在问题的

case input

3

2 3 4

1

output：

2 1 3 4

显然发生了插入错误的情况，我们要求的是在第一个大于其的数前面插入。

这里的问题在于把head指针和普通指针看作了一种东西，其实本质上没有问题，但是在插入时候就要分情况写代码，不能一概而论了。

本来也是很普通的尾插法（需要头节点+尾指针），能保证顺序不乱的插入进去。

而另一种头插法只需要头节点即可，但插入后的链表是倒序。

其实不用特殊的head指针也是可以的，但代码上要稍微冗余一点点。

针对struct和typedef struct的两种定义方法其实也没有区别，都能定义好节点，个人觉得第一种typedef显得简洁一些。

#include <stdio.h>
#include <bits/stdc++.h>
using namespace std;
/* case 2
struct node
{
    int data;
    struct node* next;
};
*/ 
typedef struct Node
{
    int data;
    Node *next;
}NODE,*node;
int main()
{
    int a, num;
    node head, p, q, t;
    head = (node)malloc(sizeof(node));
    //head = (struct node*)malloc(sizeof(struct node*)); case 2 is also ok

    //head -> next = NULL;  good habit, but not necessary.
    q = head;
    cin >> num; // original number of sequence
    for(int i = 0; i < num; ++i)
    {
        p = (node)malloc(sizeof(node)); // apply for a place of memory for node
        cin >> (p->data);

        p -> next = NULL;
        if(head->next==NULL) head->next = p; // head points at p
        else q->next = p; // make q points at p

        q = p; // 尾指针更新
    }

    cin >> a; // the one to be inserted
    t = head;

    while(t->next!= NULL && t->next->data < a) // 找到符合条件的插入点，如果都不符合则插最后
        t = t->next;
// 插入
    p = (node)malloc(sizeof(node));
    p->data = a;
    p->next = t->next;
    t->next = p;

//遍历 输出
    t = head;
    while(t->next!= NULL)
    {
        cout << t->next->data << ' ';
        t = t->next;
    }

    return 0;
}

关于内存分配

c程序中有五大内存分区：

栈：默认的存储区，一般都会用栈，由于计算机在底层对栈进行支持，其效率比较高。生长方向向下（内存地址减小），无碎片问题，但空间较小。

堆：空间大，一般pc能达到4g，同比栈区可能只有1m。分配效率较低，要手动new和delete，不然容易memory leak。生长方向向上（内存地址增大），可能会产生碎片。大量空间分配时用堆更合适。

自由存储区：用malloc分配的区域。

全局、静态存储

常量存储 const

以C为例，有几个动态内存管理的函数：

calloc(int num, int size) 返回num个size大小的 连续空间

free(void *address) 释放

malloc(int num) 一块

realloc(void *address, int newsize) 重新分配，扩展到newsize

返回值都是void*，可以随意进行指针类型转换。
